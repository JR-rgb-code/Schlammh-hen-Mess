    <script>
        // ========== GLOBAL VARIABLES ==========
        let stream = null;
        let videoElement = null;
        let canvasElement = null;
        let overlayCanvas = null;
        let overlayCtx = null;
        let cylinderRects = [];
        let calibrationData = [];
        let cylinderLabels = ['Zylinder 1', 'Zylinder 2', 'Zylinder 3'];
        let measurements = [];
        let duration = 5;
        let currentTime = 0;
        let isRunning = false;
        let timerId = null;
        let detectionThreshold = 60;
        let showMarkings = false;
        let selectedCylinderIndex = 0;

        // ========== INITIALIZATION ==========
        document.addEventListener('DOMContentLoaded', function() {
            // Initialize cylinder names
            updateCylinderNames();
            
            // Load available cameras
            loadCameras();
            
            // Set up event listeners
            document.getElementById('cylinderCount').addEventListener('change', function() {
                updateCylinderCount(parseInt(this.value));
            });
        });

        // ========== PAGE NAVIGATION ==========
        function showPage(pageId) {
            // Hide all pages
            document.querySelectorAll('.page').forEach(page => {
                page.classList.add('hidden');
            });
            
            // Show selected page
            document.getElementById(pageId).classList.remove('hidden');
            
            // Stop camera when leaving calibrate/measure pages
            if (pageId !== 'calibrate-page' && pageId !== 'measure-page') {
                stopCamera();
            }
        }

        // ========== CYLINDER CONFIGURATION ==========
        function updateCylinderCount(count) {
            count = Math.min(Math.max(count, 1), 3);
            
            // Update input value
            document.getElementById('cylinderCount').value = count;
            
            // Update cylinder labels
            const newLabels = [];
            for (let i = 0; i < count; i++) {
                newLabels.push(`Zylinder ${i + 1}`);
            }
            cylinderLabels = newLabels;
            
            // Update name inputs
            const nameInputsContainer = document.getElementById('name-inputs');
            nameInputsContainer.innerHTML = '';
            for (let i = 0; i < count; i++) {
                const input = document.createElement('input');
                input.type = 'text';
                input.className = 'cylinder-name';
                input.value = cylinderLabels[i];
                input.placeholder = 'Name';
                input.addEventListener('change', function() {
                    cylinderLabels[i] = this.value;
                });
                nameInputsContainer.appendChild(input);
            }
            
            // Update calibration data
            calibrationData = [];
            for (let i = 0; i < count; i++) {
                calibrationData.push({ top: null, bottom: null });
            }
            
            // Update cylinder rects
            if (videoElement && videoElement.videoWidth > 0) {
                cylinderRects = generateCylinderGrid(videoElement.videoWidth, videoElement.videoHeight, count);
            }
            
            // Update calibration inputs
            updateCalibrationInputs();
            
            // Update measurement tables
            updateTableHeaders();
        }

        function updateCylinderNames() {
            const count = parseInt(document.getElementById('cylinderCount').value) || 3;
            updateCylinderCount(count);
        }

        // ========== CAMERA FUNCTIONS ==========
        async function startCamera(cameraId = null) {
            try {
                // Stop existing stream
                if (stream) {
                    stream.getTracks().forEach(track => track.stop());
                }
                
                const constraints = {
                    video: {
                        deviceId: cameraId ? { exact: cameraId } : undefined,
                        width: { ideal: 1280 },
                        height: { ideal: 720 },
                        facingMode: 'environment'
                    },
                    audio: false
                };
                
                stream = await navigator.mediaDevices.getUserMedia(constraints);
                videoElement = document.getElementById('video');
                videoElement.srcObject = stream;
                
                // Wait for video to load
                videoElement.onloadedmetadata = function() {
                    const count = parseInt(document.getElementById('cylinderCount').value) || 3;
                    cylinderRects = generateCylinderGrid(videoElement.videoWidth, videoElement.videoHeight, count);
                    calibrationData = [];
                    for (let i = 0; i < count; i++) {
                        calibrationData.push({ top: null, bottom: null });
                    }
                    updateCalibrationInputs();
                    drawOverlay();
                };
                
                return true;
            } catch (error) {
                console.error('Camera error:', error);
                alert('Kamera-Zugriff fehlgeschlagen. Bitte erlauben Sie den Kamera-Zugriff in den iPhone-Einstellungen für Safari.');
                return false;
            }
        }

        function stopCamera() {
            if (stream) {
                stream.getTracks().forEach(track => track.stop());
                stream = null;
            }
            if (videoElement) {
                videoElement.srcObject = null;
            }
            if (timerId) {
                clearInterval(timerId);
                timerId = null;
            }
            isRunning = false;
        }

        async function loadCameras() {
            try {
                const devices = await navigator.mediaDevices.enumerateDevices();
                const videoDevices = devices.filter(device => device.kind === 'videoinput');
                const cameraSelect = document.getElementById('camera-select');
                
                cameraSelect.innerHTML = '';
                videoDevices.forEach(device => {
                    const option = document.createElement('option');
                    option.value = device.deviceId;
                    option.text = device.label || `Kamera ${videoDevices.indexOf(device) + 1}`;
                    cameraSelect.appendChild(option);
                });
                
                // Add default option
                const defaultOption = document.createElement('option');
                defaultOption.value = '';
                defaultOption.text = 'Standard-Kamera';
                cameraSelect.insertBefore(defaultOption, cameraSelect.firstChild);
                
                cameraSelect.addEventListener('change', async function() {
                    await startCamera(this.value);
                });
            } catch (error) {
                console.error('Error loading cameras:', error);
            }
        }

        // ========== CYLINDER FRAMES ==========
        function generateCylinderGrid(videoWidth, videoHeight, count) {
            const padding = 15;
            const availableWidth = videoWidth - padding * 2;
            const cylinderWidth = availableWidth / count;
            const cylinderHeight = videoHeight - padding * 2;
            
            return Array.from({ length: count }, (_, i) => ({
                x: padding + i * cylinderWidth,
                y: padding,
                width: cylinderWidth - 5,
                height: cylinderHeight - 5,
                label: cylinderLabels[i] || `Zylinder ${i + 1}`
            }));
        }

        function resetCylinderFrames() {
            if (videoElement && videoElement.videoWidth > 0) {
                const count = cylinderRects.length;
                cylinderRects = generateCylinderGrid(videoElement.videoWidth, videoElement.videoHeight, count);
                drawOverlay();
            }
        }

        // ========== DRAWING FUNCTIONS ==========
        function drawOverlay() {
            if (!videoElement || !videoElement.videoWidth) return;
            
            const overlay = document.getElementById('overlay-canvas');
            if (!overlay) return;
            
            overlay.width = videoElement.videoWidth;
            overlay.height = videoElement.videoHeight;
            overlayCtx = overlay.getContext('2d');
            overlayCtx.clearRect(0, 0, overlay.width, overlay.height);
            
            // Draw cylinder frames
            cylinderRects.forEach((rect, index) => {
                // Frame
                overlayCtx.strokeStyle = '#ef4444';
                overlayCtx.lineWidth = 2;
                overlayCtx.strokeRect(rect.x, rect.y, rect.width, rect.height);
                
                // Label
                overlayCtx.fillStyle = '#ffffff';
                overlayCtx.font = '16px Arial';
                const label = `${rect.label}: ${detectedML[index]?.toFixed(0) || '0'} ml`;
                const textWidth = overlayCtx.measureText(label).width;
                overlayCtx.fillRect(rect.x, rect.y - 20, textWidth + 10, 20);
                overlayCtx.fillStyle = '#000000';
                overlayCtx.fillText(label, rect.x + 5, rect.y - 5);
                
                // Fill level line (if calibrated)
                if (calibrationData[index]?.top && calibrationData[index]?.bottom && detectedML[index] > 0) {
                    const { top, bottom, topML, bottomML } = calibrationData[index];
                    const totalRange = bottomML - topML;
                    const mlFromTop = detectedML[index] - topML;
                    const ratio = mlFromTop / totalRange;
                    const fillY = top.y + ratio * (bottom.y - top.y);
                    
                    overlayCtx.strokeStyle = '#10b981';
                    overlayCtx.lineWidth = 3;
                    overlayCtx.beginPath();
                    overlayCtx.moveTo(rect.x, fillY);
                    overlayCtx.lineTo(rect.x + rect.width, fillY);
                    overlayCtx.stroke();
                    
                    // Fill area
                    overlayCtx.fillStyle = 'rgba(16, 185, 129, 0.3)';
                    overlayCtx.fillRect(rect.x, fillY, rect.width, bottom.y - fillY);
                }
                
                // Resize handle
                overlayCtx.fillStyle = '#ef4444';
                overlayCtx.fillRect(rect.x + rect.width - 10, rect.y + rect.height - 10, 10, 10);
            });
            
            // Draw calibration markings
            if (showMarkings) {
                calibrationData.forEach((cal, index) => {
                    if (cal.top) {
                        overlayCtx.fillStyle = '#3b82f6';
                        overlayCtx.beginPath();
                        overlayCtx.arc(cal.top.x, cal.top.y, 5, 0, Math.PI * 2);
                        overlayCtx.fill();
                        overlayCtx.fillStyle = '#ffffff';
                        overlayCtx.font = '12px Arial';
                        overlayCtx.fillText(`${cal.top.ml} ml`, cal.top.x + 8, cal.top.y - 5);
                    }
                    if (cal.bottom) {
                        overlayCtx.fillStyle = '#3b82f6';
                        overlayCtx.beginPath();
                        overlayCtx.arc(cal.bottom.x, cal.bottom.y, 5, 0, Math.PI * 2);
                        overlayCtx.fill();
                        overlayCtx.fillStyle = '#ffffff';
                        overlayCtx.font = '12px Arial';
                        overlayCtx.fillText(`${cal.bottom.ml} ml`, cal.bottom.x + 8, cal.bottom.y + 15);
                    }
                });
            }
            
            requestAnimationFrame(drawOverlay);
        }

        // ========== IMAGE PROCESSING ==========
        function detectHorizontalLines(imageData, rect, threshold = 60) {
            const { data, width, height } = imageData;
            const { x, y, width: rectW, height: rectH } = rect;
            const lines = [];
            
            for (let row = y + 5; row < y + rectH - 5; row++) {
                let lineStart = -1;
                let lineEnd = -1;
                let inLine = false;
                
                for (let col = x; col < x + rectW; col++) {
                    const index = (row * width + col) * 4;
                    if (index + 3 >= data.length) continue;
                    
                    const r = data[index];
                    const g = data[index + 1];
                    const b = data[index + 2];
                    const brightness = (r + g + b) / 3;
                    
                    const isDark = brightness < threshold;
                    
                    if (isDark && !inLine) {
                        lineStart = col;
                        inLine = true;
                    } else if (!isDark && inLine) {
                        lineEnd = col - 1;
                        inLine = false;
                        
                        if (lineEnd - lineStart >= 15) {
                            lines.push({
                                y: row,
                                x: lineStart,
                                length: lineEnd - lineStart,
                                brightness: brightness
                            });
                        }
                    }
                }
                
                if (inLine && lineStart !== -1) {
                    lines.push({
                        y: row,
                        x: lineStart,
                        length: rectW - (lineStart - x),
                        brightness: brightness
                    });
                }
            }
            
            lines.sort((a, b) => a.y - b.y);
            
            const filteredLines = [];
            for (let i = 0; i < lines.length; i++) {
                if (i === 0 || lines[i].y - lines[i-1].y > 8) {
                    filteredLines.push(lines[i]);
                }
            }
            
            return filteredLines;
        }

        function detectSludgeLevelML(imageData, cylinderRects, calibrationData) {
            const results = [];
            
            for (let i = 0; i < cylinderRects.length; i++) {
                const rect = cylinderRects[i];
                const calibration = calibrationData[i];
                
                if (!calibration || !calibration.top || !calibration.bottom) {
                    // Without calibration, estimate based on image
                    const lines = detectHorizontalLines(
                        { data: imageData.data, width: imageData.width, height: imageData.height },
                        rect,
                        detectionThreshold
                    );
                    
                    if (lines.length > 0) {
                        const sludgeY = lines[lines.length - 1].y;
                        const heightPercent = (sludgeY - rect.y) / rect.height;
                        results.push(100 - heightPercent * 100);
                    } else {
                        results.push(0);
                    }
                } else {
                    // With calibration
                    const { top, bottom, topML, bottomML } = calibration;
                    let sludgeY = rect.y + rect.height;
                    
                    for (let row = top.y; row < bottom.y; row++) {
                        let isWater = false;
                        let isSludge = false;
                        
                        for (let col = rect.x; col < rect.x + rect.width; col += Math.max(1, Math.floor(rect.width / 5))) {
                            const index = (row * imageData.width + col) * 4;
                            if (index + 3 >= imageData.data.length) continue;
                            
                            const r = imageData.data[index];
                            const g = imageData.data[index + 1];
                            const b = imageData.data[index + 2];
                            const brightness = (r + g + b) / 3;
                            
                            if (brightness < 80) {
                                isSludge = true;
                            } else {
                                isWater = true;
                            }
                        }
                        
                        if (isWater && isSludge) {
                            sludgeY = row;
                            break;
                        }
                    }
                    
                    const totalHeight = bottom.y - top.y;
                    const heightFromTop = sludgeY - top.y;
                    const ratio = heightFromTop / totalHeight;
                    const mlValue = topML + (bottomML - topML) * ratio;
                    
                    results.push(mlValue);
                }
            }
            
            return results;
        }

        function testDetection() {
            if (!videoElement || !videoElement.videoWidth) {

        function autoDetectMarkings() {
            if (!videoElement || !videoElement.videoWidth) {
                alert('Bitte starte zuerst die Kamera.');
                return;
            }
            
            const rect = cylinderRects[selectedCylinderIndex];
            const canvas = document.createElement('canvas');
            canvas.width = videoElement.videoWidth;
            canvas.height = videoElement.videoHeight;
            const ctx = canvas.getContext('2d');
            ctx.drawImage(videoElement, 0, 0, canvas.width, canvas.height);
            
            const imageData = ctx.getImageData(rect.x, rect.y, rect.width, rect.height);
            const lines = detectHorizontalLines(
                { data: imageData.data, width: rect.width, height: rect.height },
                { x: 0, y: 0, width: rect.width, height: rect.height },
                detectionThreshold
            );
            
            if (lines.length >= 2) {
                const topLine = lines[0];
                const bottomLine = lines[lines.length - 1];
                
                const topML = prompt(`Oberste Markierung (Y=${topLine.y}): Welcher ml-Wert?`, "100");
                if (topML === null) return;
                
                const bottomML = prompt(`Unterste Markierung (Y=${bottomLine.y}): Welcher ml-Wert?`, "0");
                if (bottomML === null) return;
                
                setCalibrationPoint(
                    selectedCylinderIndex,
                    'top',
                    rect.x + rect.width / 2,
                    rect.y + topLine.y,
                    parseFloat(topML)
                );
                setCalibrationPoint(
                    selectedCylinderIndex,
                    'bottom',
                    rect.x + rect.width / 2,
                    rect.y + bottomLine.y,
                    parseFloat(bottomML)
                );
                
                updateCalibrationInputs();
                checkAllCalibrated();
            } else {
                alert('Nicht genug Markierungen erkannt. Versuche es mit manueller Kalibrierung oder passe den Schwellenwert an.');
            }
        }

        function setCalibrationPoint(index, pointType, x, y, mlValue) {
            calibrationData[index][pointType] = { x, y, ml: mlValue };
            updateCalibrationInputs();
            checkAllCalibrated();
        }

        function updateCalibrationInputs() {
            const container = document.getElementById('calibration-inputs');
            container.innerHTML = '';
            
            cylinderRects.forEach((rect, index) => {
                const cylinderDiv = document.createElement('div');
                cylinderDiv.className = 'mb-3';
                
                const title = document.createElement('h4');
                title.className = 'font-medium mb-2';
                title.textContent = rect.label;
                cylinderDiv.appendChild(title);
                
                const topDiv = document.createElement('div');
                topDiv.className = 'mb-2';
                topDiv.innerHTML = `
                    <label class="text-sm text-gray-600">Oberste Markierung (ml)</label>
                    <input type="number" class="calibration-top" data-index="${index}" 
                           value="${calibrationData[index]?.top?.ml || ''}" 
                           placeholder="z.B. 100" style="width: 100%; padding: 8px; border: 1px solid #d1d5db; border-radius: 4px;">
                `;
                cylinderDiv.appendChild(topDiv);
                
                const bottomDiv = document.createElement('div');
                bottomDiv.className = 'mb-2';
                bottomDiv.innerHTML = `
                    <label class="text-sm text-gray-600">Unterste Markierung (ml)</label>
                    <input type="number" class="calibration-bottom" data-index="${index}" 
                           value="${calibrationData[index]?.bottom?.ml || ''}" 
                           placeholder="z.B. 0" style="width: 100%; padding: 8px; border: 1px solid #d1d5db; border-radius: 4px;">
                `;
                cylinderDiv.appendChild(bottomDiv);
                
                const setButton = document.createElement('button');
                setButton.className = 'btn btn-success';
                setButton.textContent = 'Positionen automatisch setzen';
                setButton.style.width = '100%';
                setButton.style.marginTop = '8px';
                setButton.onclick = function() {
                    const topML = parseFloat(topDiv.querySelector('input').value) || 100;
                    const bottomML = parseFloat(bottomDiv.querySelector('input').value) || 0;
                    
                    if (!topML || !bottomML) {
                        alert('Bitte gib beide ml-Werte ein.');
                        return;
                    }
                    
                    setCalibrationPoint(
                        index,
                        'top',
                        rect.x + rect.width / 2,
                        rect.y + 20,
                        topML
                    );
                    setCalibrationPoint(
                        index,
                        'bottom',
                        rect.x + rect.width / 2,
                        rect.y + rect.height - 20,
                        bottomML
                    );
                };
                cylinderDiv.appendChild(setButton);
                
                // Add success message if calibrated
                if (calibrationData[index]?.top && calibrationData[index]?.bottom) {
                    const successDiv = document.createElement('div');
                    successDiv.className = 'success-message';
                    successDiv.textContent = `✅ Kalibriert: ${calibrationData[index].top.ml} ml – ${calibrationData[index].bottom.ml} ml`;
                    cylinderDiv.appendChild(successDiv);
                }
                
                container.appendChild(cylinderDiv);
            });
            
            // Add event listeners
            document.querySelectorAll('.calibration-top').forEach(input => {
                input.addEventListener('change', function() {
                    const index = parseInt(this.getAttribute('data-index'));
                    calibrationData[index].top = { ...calibrationData[index].top, ml: parseFloat(this.value) };
                });
            });
            
            document.querySelectorAll('.calibration-bottom').forEach(input => {
                input.addEventListener('change', function() {
                    const index = parseInt(this.getAttribute('data-index'));
                    calibrationData[index].bottom = { ...calibrationData[index].bottom, ml: parseFloat(this.value) };
                });
            });
        }

        function startMeasurement() {
            duration = parseInt(document.getElementById('duration').value) || 5;
            currentTime = 0;
            measurements = [];
            isRunning = true;
            
            // Take first measurement
            testDetection();
            addMeasurement(detectedML);
            
            // Set up timer
            timerId = setInterval(() => {
                currentTime++;
                testDetection();
                addMeasurement(detectedML);
                
                document.getElementById('current-time').textContent = `${currentTime}/${duration} min`;
                
                if (currentTime >= duration) {
                    stopMeasurement();
                }
            }, 60000);
            
            showPage('measure-page');
            updateMeasureCamera();
            updateCurrentMLValues();
        }

        function exportCSV() {
            const headers = ["Zeit (min)", ...cylinderLabels.map(l => `${l} (ml)`), "Timestamp"];
            const rows = measurements.map(m => [
                m.time,
                ...m.mlValues,
                new Date(m.timestamp).toLocaleString('de-DE')
            ]);
            
            const csvContent = [
                headers.join(","),
                ...rows.map(row => row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(","))
            ].join("\n");
            
            const blob = new Blob([csvContent], { type: "text/csv;charset=utf-8;" });
            const url = URL.createObjectURL(blob);
            const link = document.createElement("a");
            link.href = url;
            link.download = `sedimentation_ml_${new Date().toISOString().slice(0, 10)}.csv`;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
